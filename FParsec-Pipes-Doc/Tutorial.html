type Date = { Year : int; Month : int; Day : int }

let dateShape : Parser<_, unit> =
    !- (digits 4) -- '-' -- digits 2 -- '-' -- digits 2 -|> ()

let date : Parser<Date, unit> =
    !+ (digits 4) -- '-' -+ digits 2 -- '-' -+ digits 2
    -|> fun yyyy mm dd -> { Year = yyyy; Month = mm; Day = dd }
